\noindent\textbf{Исходное уравнение:}\\
\section*{Использованные правила}
\begin{itemize}
    \item $(u + v)' = u' + v'$
    \item $(u - v)' = u' - v'$
    \item $(uv)' = u'v + uv'$
    \item $\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}$
    \item $(u^n)' = n \cdot u^{n-1} \cdot u'$
    \item $(\sin u)' = \cos u \cdot u'$
    \item $(\cos u)' = -\sin u \cdot u'$
    \item $(\tan u)' = \frac{u'}{\cos^2 u}$
    \item $(\ln u)' = \frac{u'}{u}$
\end{itemize}
